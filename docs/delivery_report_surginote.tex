\section{مقدمه و تطابق با دستورکار تسک}
طبق دستورکار تسک (نسخه متنی موجود در مخزن / فایل \lr{PDF}) باید پیاده‌شود:\\
\textbf{(۱) ورود و ذخیره:} اندپوینت بارگذاری، پارس و اعتبارسنجی، نگاشت، ذخیرهٔ \lr{raw} و استخراج سطرهای طبیعی‌شدهٔ پنل‌های ویدیو، فازها، مهارت‌ها و کامنت‌ها.\\
\textbf{(۲) تحلیل اولیه:} میانگین نمرهٔ هر مهارت؛ منطق تناقض «نمره بالا با مارکر منفی/هشدار».\\
\textbf{(۳) چند مرحله در یک \lr{JSON}:} خلاصه فازها؛ تحلیل نمره با متن قالبی؛ خط زمانی کامنت‌ها؛ بخش پرچم‌ها.\\
\textbf{(۴) بدون ابزار پیچیده یا برنامه‌ریزی عاملی.}

\begin{mybox}
\textbf{خلاصه نتیجه:} هر چهار مورد به‌صورت ماژول \lr{\texttt{app/application/report\_pipeline.py}} و \lr{\texttt{app/application/analytics.py}} و اندپوینت‌های زیر مجموعهٔ \lr{\texttt{/v1}} پیاده کرده‌شد؛ روایت با \lr{Gemini} \emph{جدای از الزام MVP} است اما آمادهٔ تولید و تست واقعی با کلید \lr{.env}.
\end{mybox}

\section{استک تکنیکی و نقش هر جزء}
\begin{itemize}
  \item \textbf{\lr{FastAPI}} — لایهٔ \lr{HTTP}، بازآموزندهٔ باز \lr{(OpenAPI 3.1)}، اعتبارسنجی ورودی/خروج با \lr{Pydantic}.
  \item \textbf{\lr{SQLite} + SQLAlchemy} — مخزن \lr{cases} با \lr{raw\_payload}؛ جداول فازها، مهارت‌ها، کامنت‌ها، گزارش‌های ذخیره‌شده.
  \item \textbf{\lr{pandas + openpyxl}} — پارس پنج شیٔنام استاندارد \lr{xlsx}، فیلتر سطرهای جمع مهارت‌ها.
  \item \textbf{\lr{httpx}} — کلاینت \lr{REST} خام به \lr{generativelanguage.googleapis.com/v1beta} برای \lr{Gemini}.
  \item \textbf{\lr{pydantic-settings}} — خواندن \lr{.env} از ریشهٔ قطعی ریپو (نه الزاماً \lr{cwd} پوسته)، پیشوند \lr{SN\_*} و کلیدهای \lr{GEMINI\_*}.
\end{itemize}

\section{لایه‌ها (معماری داخل مخزن)}
\begin{enumerate}
  \item \textbf{\lr{domain}} — ارورهای دامنه، بدون وابستگی به فریمورک.
  \item \textbf{\lr{application}} — \lr{ingest.py}، \lr{analytics.py}، \lr{report\_pipeline.py}، \lr{narrative.py}.
  \item \textbf{\lr{infrastructure}} — پارسر اکسل، مدل‌ها، سشن، مخزن، \lr{llm/gemini\_rest.py}.
  \item \textbf{\lr{api}} — \lr{router.py}، \lr{schemas.py}، \lr{locale\_util.py}.
\end{enumerate}

\section{محتوای اصلی \lr{JSON} گزارش (نسخه \lr{1.3.0})}
\begin{itemize}
  \item \texttt{meta}: شامل \textbf{\texttt{schema\_version}}, \textbf{\texttt{report\_locale}}, نام ویدئو، طول مدت زمان‌ها، ورژن خروجی، زمان ساخت UTC، خط‌مشی پرچم و آستانه نسبتی.
  \item \texttt{sections.phase\_summary.items}: هر فاز با بازهٔ نمایش زمان، فریم‌ها، توضیح، روش احتمالاً فاکویی؛ داده خام بالینی انگلیسی از ابزار باقی می‌ماند.
  \item \texttt{sections.score\_analysis}: میانگین مهارت هر فاز؛ \texttt{score\_narrative} به زبان خواست‌شدهٔ قالب.
  \item \texttt{sections.comments\_timeline}: مرتب‌شدهٔ زمانی، زمان‌ها حفظ.
  \item \texttt{sections.contradictions.flags}: شامل کد انگلیسی \lr{(برای پارسرها)}؛ یاداشت‌های انسانی روی بعضی پیوندهها در \texttt{locale=fa}.
  \item \texttt{quality.limitations}: سطرهای حقوق بالینی عمومی فارسی یا انگلیسی بسته به \texttt{report\_locale}.
\end{itemize}

\section{مشکلاتی که واقعاً پیش آمدند و تصمیم مهندسی}
\begin{itemize}
  \item بارگذاری \lr{.env} وابسته به \lr{cwd} بود ⇒ رفع با مسیر قطعی کنار ماژول \lr{config}.
  \item \lr{PowerShell Invoke-RestMethod + ConvertTo-Json} گاهی بدنهٔ پارس‌شدنی نشان نمی‌داد ⇒ اندپوینت کمکی \textbf{\lr{/v1/narratives/generate-from-report}} تا بدنه فقط کل گزارش باشد؛ کوئری برای گزینه‌ها.
  \item ظرف مدل؛ مدل پیش‌فرض ریپو به \textbf{\lr{gemini-2.5-flash}} نزد به‌روز واقعی جهت کمینه‌کردن خطای \lr{NOT\_FOUND}.
  \item بین جملات غیر فارسی موجود در ستون انگلیسی خروجی ابزار و فارسیٔ قالب داخلی مخلوط پیش می‌آید؛ فرانت خودش برچسب زبان جدا نشان می‌دهد.
\end{itemize}
